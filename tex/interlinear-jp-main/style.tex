      \clearpage
    }{%
      \pagecolor{CoverGround}\color{CoverGold}%
      \vspace*{0.18\paperheight}
      {\centering{\jpfont\fontsize{28pt}{31pt}\selectfont #1\par}\vspace{0.5em}{\zhfont\fontsize{14pt}{17pt}\selectfont #2\par}\vspace{1.4em}{\jpfont\fontsize{9pt}{12pt}\selectfont #3\par}}

      \clearpage\nopagecolor\color{BookInk}%
    }%
  \fi
  \restoregeometry
  \thispagestyle{empty}
  \vspace*{0.20\textheight}
