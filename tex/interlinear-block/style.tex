\NewDocumentCommand{\InterTitle}{m m m m}{%
  \frontmatter
  \thispagestyle{empty}
  \if\relax\detokenize{#4}\relax
  \else
    \newgeometry{margin=0pt}%
    \IfFileExists{#4}{%
      \noindent\includegraphics[width=\paperwidth,height=\paperheight]{#4}\par
      \clearpage
    }{}%
    \restoregeometry
  \fi
  \thispagestyle{empty}
  \vspace*{0.16\textheight}
  {\centering{\zhfont\Huge #1\par}\vspace{0.8em}{\jpfont\small #2\par}\vspace{1.1em}{\jpfont\normalsize #3\par}\vfill\GrammarLegend\par}
  \clearpage
  \mainmatter
}
