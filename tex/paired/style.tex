\newcommand{\PairLabel}[1]{%
  {\sffamily\bfseries\fontsize{6pt}{6pt}\selectfont\textcolor{BookLabel}{#1}}%
}
\newcommand{\MainZH}[1]{{\zhfont\fontsize{10.2pt}{15.8pt}\selectfont\color{BookInk}#1}}
\newcommand{\MainJP}[1]{{\jpfont\fontsize{9.2pt}{15.2pt}\selectfont\color{BookInk}#1}}
\newcommand{\CommentText}[1]{{\commentfont\fontsize{7.3pt}{11.6pt}\selectfont\color{BookMuted}#1}}

\newcommand{\MaybeComment}[2]{%
  \if\relax\detokenize{#2}\relax
  \else
    \par\vspace{0.16em}\noindent\PairLabel{#1}\hspace{0.45em}\CommentText{#2}%
  \fi
}

\NewDocumentCommand{\Pair}{m m m m}{%
  \Needspace{7\baselineskip}%
  \par\vspace{0.25em}%
  \noindent\color{BookRule}\rule{\linewidth}{0.25pt}\par\nobreak\vspace{0.28em}%
  \ifPairSideBySide
    \noindent\begin{tabularx}{\linewidth}{@{}>{\raggedright\arraybackslash}X@{\hspace{0.65em}}>{\raggedright\arraybackslash}X@{}}
      \PairLabel{中}\hspace{0.35em}\MainZH{#1} &
      \PairLabel{日}\hspace{0.35em}\MainJP{#2}
    \end{tabularx}%
    \MaybeComment{注}{#3}%
    \MaybeComment{註}{#4}%
  \else
    \noindent\PairLabel{中}\hspace{0.55em}\MainZH{#1}\par
    \vspace{0.16em}%
    \noindent\PairLabel{日}\hspace{0.55em}\MainJP{#2}%
    \MaybeComment{注}{#3}%
    \MaybeComment{註}{#4}%
  \fi
  \par\vspace{0.35em}%
}
