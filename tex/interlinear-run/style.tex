\NewDocumentCommand{\RubyFixed}{m m}{%
  \ifShowRuby
    \leavevmode
    \begingroup
      \setbox0=\hbox{#1}%
      \dimen0=\wd0
      \vbox{%
        \offinterlineskip
        \hbox to \dimen0{\hss{\RubyFont #2}\hss}%
        \kern0.06ex
        \box0
      }%
    \endgroup
  \else
    #1%
  \fi
}
\NewDocumentCommand{\RubyWide}{m m}{%
  \ifShowRuby
    \leavevmode
    \begingroup
      \setbox0=\hbox{#1}%
      \setbox1=\hbox{{\RubyFont #2}}%
      \dimen0=\wd0
      \ifdim\wd1>\dimen0 \dimen0=\wd1\fi
      \vbox{%
        \offinterlineskip
        \hbox to \dimen0{\hss\box1\hss}%
        \kern0.06ex
        \hbox to \dimen0{\hss\box0\hss}%
      }%
    \endgroup
  \else
    #1%
  \fi
}

\newdimen\RunUnitWidth
\NewDocumentCommand{\RunUnit}{m m m}{%
  \begingroup
    \setbox0=\hbox{\ZHRun{#1}}%
    \RunUnitWidth=\wd0
    \ifdim\RunUnitWidth>\linewidth
      \RunUnitWidth=\linewidth
    \fi
    \leavevmode
    \vtop{%
      \hsize=\RunUnitWidth
      \parindent=0pt
      \rightskip=0pt plus 1fil
      \ZHRun{#1}\par
      \vspace{-0.08em}
      \JARunGloss{#2}\par
      \if\relax\detokenize{#3}\relax\else
        \vspace{-0.18em}
        {\TinyLabel{注}}\hspace{0.2em}\JARunExplain{#3}\par
      \fi
      \vspace{0.12em}
    }%
    \penalty0\hskip0pt plus 0.25em\relax
  \endgroup
}
